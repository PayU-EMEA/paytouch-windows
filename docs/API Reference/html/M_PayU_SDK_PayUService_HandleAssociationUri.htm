Parameters

uri
Type: SystemUri

Return Value

Type: Boolean
Returns true if PayTouch handled properly Uri. Returns false if PayTouch can't handle provided Uri.
Examples
Copy
// In WMAppManifest.xml add URI association:
// 
// <Extensions>
//    <Protocol Name="merchant-sample-app-protocol" NavUriFragment="encodedLaunchUri=%s" TaskID="_default" />
// </Extensions>

public partial class App : Application
{
   private void InitializePhoneApplication()
   {
       RootFrame = new PhoneApplicationFrame();
       RootFrame.UriMapper = new AssociationUriMapper();

       PayUService.Instance.MerchantAppProtocol = "merchant-sample-app-protocol";
   }

   private class AssociationUriMapper : UriMapperBase
   {
       public override Uri MapUri(Uri uri)
       {
           if (PayUService.Instance.HandleAssociationUri(uri))
           {
               // Uri was properly handled by PayTouch.
               // 
               return new Uri("/MainPage.xaml", UriKind.Relative);
           }
           else
           {
               // PayTouch can't handle provided Uri.
           }

           return uri;
       }
   }
}
